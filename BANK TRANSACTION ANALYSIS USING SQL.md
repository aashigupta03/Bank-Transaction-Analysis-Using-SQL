\# DITINCT CATEGORIES AND THEIR COUNT

SELECT DISTINCT(CATEGORY), COUNT(CATEGORY) AS NUMBER FROM mytransaction

GROUP BY CATEGORY

ORDER BY NUMBER DESC;



\# TOP 10 HIGHEST WITHDRAWALS

SELECT transaction\_id, Withdrawal FROM mytransaction

ORDER BY Withdrawal DESC

LIMIT 10;



\# WHICH CATEGORY CONSUMES MORE MONEY THAN 20000?

SELECT CATEGORY, SUM(WITHDRAWAL) AS Total\_Spent FROM mytransaction

GROUP BY CATEGORY

HAVING Total\_Spent>20000

ORDER BY Total\_Spent DESC;



\# INCOME VS EXPENSE

SELECT SUM(DEPOSIT) AS INCOME,

SUM(WITHDRAWAL) AS EXPENSE,

SUM(DEPOSIT)-SUM(WITHDRAWAL) AS NET\_SAVINGS

FROM mytransaction;



\# HIGHEST DEPOSIT

SELECT \* FROM MYTRANSACTION

ORDER BY DEPOSIT DESC

LIMIT 1;



\# AVERAGE WITHDRWAL

SELECT AVG(WITHDRAWAL) AS AVERAGE\_WITHDRAWAL FROM MYTRANSACTION

WHERE WITHDRAWAL>0;



\# TOP 5 CATEGORIES BY SPENDING

SELECT CATEGORY, SUM(WITHDRAWAL) AS TOTAL\_SPENDING FROM mytransaction

GROUP BY CATEGORY

ORDER BY TOTAL\_SPENDING DESC

LIMIT 5;



\# Category Wise Average Spending

SELECT CATEGORY, AVG(WITHDRAWAL) AS AVG\_SPENDING

FROM mytransaction

GROUP BY CATEGORY;



\# RANK CATEGORIES BY SPENDING

SELECT CATEGORY, WITHDRAWAL,

RANK() OVER (ORDER BY WITHDRAWAL DESC) AS RANKING

FROM mytransaction;



\# RUNNING BALANCE TREND

SELECT DATE, BALANCE,

LAG(BALANCE) OVER(ORDER BY DATE) AS PREVIOUS\_DEPOSIT

FROM mytransaction;



\# PRESENT DAY BALANCE VS PREVIOUS DAY BALANCE 

SELECT DATE, BALANCE AS CURRENT\_BALANCE,

LAG(BALANCE) OVER(ORDER BY DATE) AS PREVIOUS\_BALANCE,

BALANCE-LAG(BALANCE) OVER(ORDER BY DATE) AS DIFFERENCE

FROM mytransaction;



\# Percentage Contribution of Each Category

SELECT CATEGORY,ROUND(SUM(WITHDRAWAL)\*100/(SELECT SUM(WITHDRAWAL) 

FROM mytransaction),2)

AS SPENDING\_PERCENTAGE FROM mytransaction

GROUP BY CATEGORY

ORDER BY SPENDING\_PERCENTAGE DESC;



\# DETECT UNUSUALLY LARGE EXPENSES

SELECT \* FROM mytransaction WHERE 

WITHDRAWAL>

(SELECT AVG(WITHDRAWAL)\*3 FROM mytransaction);



\# Top 3 SPENDING COUNTRIES WITH RANK

SELECT CATEGORY, SUM(WITHDRAWAL), 

RANK() OVER(ORDER BY SUM(WITHDRAWAL) DESC) AS RANKING 

FROM mytransaction

GROUP BY CATEGORY

LIMIT 3;



\# HIGH/MEDIUM/LOW EXPENSE LABEL 

SELECT TRANSACTION\_ID,CATEGORY,WITHDRAWAL,

CASE

WHEN WITHDRAWAL>5000 THEN 'HIGH EXPENSE'

WHEN WITHDRAWAL BETWEEN 1000 AND 4999 THEN 'MEDIUM EXPENSE'

WHEN WITHDRAWAL>0 THEN 'LOW EXPENSE'

ELSE 'NO EXPENSE'

END AS EXPENSE\_LABEL

FROM mytransaction;



\# SPENDING BEHAVIOUR CATEGORY WISE 

SELECT  CATEGORY, SUM(WITHDRAWAL) AS TOTAL\_SPEND,

CASE 

WHEN SUM(WITHDRAWAL)>20000 THEN "HIGH SPEND"

WHEN SUM(WITHDRAWAL) BETWEEN 10000 AND 19999 THEN 'MEDIUM SPEND'

ELSE 'LOW SPEND'

END AS SPENDING\_CATEGORY

FROM mytransaction

GROUP BY CATEGORY;



\# Customer Financial Behavior Profile

SELECT CATEGORY, SUM(DEPOSIT) AS TOTAL\_DEPOSIT,

SUM(WITHDRAWAL) AS TOTAL\_SPENDING,

CASE 

WHEN SUM(DEPOSIT)-SUM(WITHDRAWAL)>10000 THEN 'SAVING CUSTOMER'

WHEN SUM(DEPOSIT)-SUM(WITHDRAWAL) BETWEEN 0 AND 10000 THEN 'BALANCED 

CUSTOMER'

ELSE 'HIGH SPENDER'

END AS CUSTOMER\_TYPE

FROM mytransaction

GROUP BY CATEGORY;
